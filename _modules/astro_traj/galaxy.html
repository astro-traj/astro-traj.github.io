    def __init__(self, Mspiral, Mbulge, Mhalo, R_eff, rcut=100.0):
        """
        Galaxy class. Masses in Msun, distances in kpc. Immediately converted to SI. 
        """
        self.Mspiral = Mspiral*C.M_sun.value
        self.Mbulge = Mbulge*C.M_sun.value
        self.Mhalo = Mhalo*C.M_sun.value
        self.R_eff = R_eff*C.kpc.value
        self.rcut = rcut*C.kpc.value                 # distance at which potential should go to 0

        self.abulge = self.R_eff / 1.8153             # Use relationship between R_effective and abulge from Hernquist 1990 (eq. 38)
        self.rcore = 100.0                            # Used in Belczynski+2006 halo potential, should be where DM density profile flattens FIXME
    




[docs]    def Miyamaoto_Nagai_Hernquist(self):
        """
        Galaxy with Hernquist potential for stellar component and dark matter potential from Miyamaoto & Nagai 1975
        """

        #FIXME: Need to make sure U(r>rcut) is implemented properly. I think it's right, but I want someone else's eyes on it
        def Uhalo(r):
            """
            Galaxy class
            """

            G = C.G.value
            if r>rcut:
                return rcut*Uhalo(rcut)/r
            else:
                term1 = .5*np.log(1+((r/rcore)**2))
                term2 = rcore*np.arctan(r/rcore)/r
                return -1.0*G*Mhalo*(term1+term2)/rcore
        def Ubulge(r):
            """
            Galaxy class
            """
            G=C.G.value


            return -G*Mbulge/(r+abulge)

        def Vdot(time,COORD):
            """
            Galaxy class
            """
            G=C.G.value


            x,y,z = COORD[3:]
            xdot,ydot,zdot = COORD[:3]

            r = np.sqrt((x**2)+(y**2)+(z**2))

            def halo(r_i,r): #gradient of equation 6 in BELCZYNSKI, BULIK, & RUDAK 2002 Apj 571, 394
                """
                Galaxy class
                """
                term1 = r_i/((rcore**2)+(r**2))
                term2 = r_i/((r**2)*(1.0+((r/rcore)**2)))
                term3 = -1.0*r_i*rcore*np.arctan(r/rcore)/(r**3)

                if r>rcut:
                    #print 'r>rcut'#gradient of halo potential past cutoff where it goes like 1/r
                    P = rcut*self.Uhalo(rcut)
                    return -P*r_i/(r**1.5)


                else: return -G*Mhalo*(term1+term2+term3)/rcore


            def bulge(r_i,r): #gradient of equation found in Section 2 of Belczynski et al. 2006 Apj 648,1110
                """
                Galaxy class
                """
                return G*Mbulge*r_i/(r*((abulge+r)**2))


            gradUx = halo(x,r)+bulge(x,r)
            gradUy = halo(y,r)+bulge(y,r)
            gradUz = halo(z,r)+bulge(z,r)

            return np.array([-gradUx,-gradUy,-gradUz, xdot, ydot, zdot])


        self.Vdot = Vdot
        self.Ubulge = Ubulge
        self.Uhalo = Uhalo
